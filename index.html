Dark Mode Toggle
5% баллов — самая простая обязательная фича

💡 Принцип работы

При клике на кнопку добавляем/убираем класс dark на тег body. CSS сам меняет цвета.

CSS 
/* Светлая тема — по умолчанию */
body {
  background: #ffffff;
  color: #000000;
}

/* Тёмная тема — когда есть класс dark */
body.dark {
  background: #121212;
  color: #ffffff;
}
JavaScript 
const btn = document.querySelector('#darkBtn');

btn.addEventListener('click', function() {
  document.body.classList.toggle('dark');
  
  // Меняем текст кнопки
  if (document.body.classList.contains('dark')) {
    btn.textContent = '☀️ Light Mode';
  } else {
    btn.textContent = '🌙 Dark Mode';
  }
});

📝 Твоё задание

  • Создай страницу с кнопкой "Dark Mode"
  • При клике меняй фон и цвет текста
  • Кнопка должна менять текст: "Dark Mode" ↔ "Light Mode"
  • Бонус: сохраняй выбор в localStorage

Modal окно
5% баллов — появляется при клике на карточку

💡 Принцип работы

Modal — это скрытый div поверх всей страницы. По клику убираем класс hidden, заполняем данными.

HTML 
<!-- Оверлей и модальное окно -->
<div id="modal" class="modal hidden">
  <div class="modal-content">
    <button id="closeBtn">✕</button>
    <h2 id="modal-title"></h2>
    <p id="modal-desc"></p>
  </div>
</div>
CSS 
.modal {
  position: fixed;
  inset: 0;           /* top:0 right:0 bottom:0 left:0 */
  background: rgba(0,0,0,0.7);
  display: flex;
  align-items: center;
  justify-content: center;
  z-index: 999;
}

.modal.hidden { display: none; }

.modal-content {
  background: white;
  padding: 30px;
  border-radius: 12px;
  max-width: 500px;
  width: 90%;
}
JavaScript 
const modal = document.getElementById('modal');
const closeBtn = document.getElementById('closeBtn');

// Открыть modal с данными карточки
function openModal(item) {
  document.getElementById('modal-title').textContent = item.name;
  document.getElementById('modal-desc').textContent = item.description;
  modal.classList.remove('hidden');
}

// Закрыть modal
closeBtn.addEventListener('click', function() {
  modal.classList.add('hidden');
});

// Закрыть при клике на фон
modal.addEventListener('click', function(e) {
  if (e.target === modal) {
    modal.classList.add('hidden');
  }
});

📝 Твоё задание

  • Создай 3 карточки с кнопкой "Подробнее"
  • При клике modal открывается с данными этой карточки
  • Кнопка ✕ закрывает modal
  • Клик на фон тоже закрывает

Search / Filter
5% баллов — фильтрация карточек по вводу

💡 Принцип работы

Слушаем событие input на поле поиска. При каждом нажатии фильтруем массив и перерисовываем карточки.

JavaScript 
const searchInput = document.getElementById('search');
const container = document.getElementById('container');

// Массив данных
const items = [
  { name: 'Tokyo', country: 'Japan', price: 1200 },
  { name: 'Paris', country: 'France', price: 900 },
  { name: 'Almaty', country: 'Kazakhstan', price: 300 },
];

// Функция рендера карточек
function renderCards(data) {
  container.innerHTML = '';
  data.forEach(function(item) {
    const card = document.createElement('div');
    card.innerHTML = `<h3>${item.name}</h3><p>${item.country}</p>`;
    container.appendChild(card);
  });
}

// Слушаем ввод
searchInput.addEventListener('input', function() {
  const query = searchInput.value.toLowerCase();
  
  const filtered = items.filter(function(item) {
    return item.name.toLowerCase().includes(query);
  });
  
  renderCards(filtered);
});

// Показать все при загрузке
renderCards(items);
⚠️ Ключевые методы: filter() — создаёт новый массив из подходящих элементов. includes() — проверяет есть ли строка внутри другой. toLowerCase() — делает всё строчными буквами чтобы регистр не важен.

📝 Твоё задание

  • Создай массив из 5 объектов (любая тема)
  • Отрисуй их как карточки
  • Добавь input и фильтруй по названию
  • Бонус: фильтруй по цене (меньше/больше числа)

Carousel / Slider
5% баллов — самая сложная фича

💡 Принцип работы

Храним текущий индекс в переменной. Кнопки prev/next меняют индекс и перерисовывают слайд.

HTML 
<div class="carousel">
  <button id="prev">◀</button>
  <div id="slide"></div>
  <button id="next">▶</button>
</div>
JavaScript 
const items = [
  { name: 'Tokyo', img: 'tokyo.jpg' },
  { name: 'Paris', img: 'paris.jpg' },
  { name: 'Dubai', img: 'dubai.jpg' },
];

let currentIndex = 0;  // текущий слайд

function showSlide() {
  const item = items[currentIndex];
  document.getElementById('slide').innerHTML = `
    <img src="${item.img}" />
    <h3>${item.name}</h3>
  `;
}

document.getElementById('next').addEventListener('click', function() {
  currentIndex++;
  if (currentIndex >= items.length) {
    currentIndex = 0;  // возврат к первому
  }
  showSlide();
});

document.getElementById('prev').addEventListener('click', function() {
  currentIndex--;
  if (currentIndex < 0) {
    currentIndex = items.length - 1;  // переход к последнему
  }
  showSlide();
});

showSlide();  // показать первый слайд сразу
⚠️ Самое важное: currentIndex — это просто число. Next: +1, Prev: -1. Если вышли за границы — возвращаемся на другой конец.

📝 Твоё задание

  • Создай массив из 4 объектов
  • Сделай кнопки prev и next
  • Показывай name и description текущего элемента
  • Бонус: добавь точки-индикаторы (●○○○)

💡 Структура data.json

На экзамене дадут data.json. Вот типичная структура для любой темы:

data.json (Travel пример) 
[
  {
    "id": 1,
    "name": "Tokyo",
    "country": "Japan",
    "price": 1200,
    "description": "Amazing city with rich culture",
    "image": "assets/tokyo.jpg",
    "tags": ["culture", "food", "modern"]
  },
  {
    "id": 2,
    "name": "Paris",
    "country": "France",
    "price": 900,
    "description": "City of love and art",
    "image": "assets/paris.jpg",
    "tags": ["art", "romance", "history"]
  }
]
JavaScript — рендер карточек 
const container = document.getElementById('cards-container');

function renderCards(data) {
  container.innerHTML = '';  // очищаем перед рендером
  
  data.forEach(function(item) {
    const card = document.createElement('div');
    card.className = 'card';
    
    card.innerHTML = `
      <img src="${item.image}" alt="${item.name}" />
      <h3>${item.name}</h3>
      <p>${item.country}</p>
      <p>$${item.price}</p>
      <button class="details-btn">Подробнее</button>
    `;
    
    // Клик на карточку открывает modal
    card.addEventListener('click', function() {
      openModal(item);
    });
    
    container.appendChild(card);
  });
}

// Вызываем при загрузке страницы
renderCards(destinations);
⚠️ На ГОС данные будут в data.json файле. Так как интернета нет — не нужен fetch. Просто скопируй данные из data.json в JS переменную: const destinations = [ ... ]

.cards-grid {
